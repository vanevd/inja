130  bool parse_expression_datum(Template& tmpl) {
131  nonstd::string_view json_first;
132  size_t bracket_level = 0;
133  size_t brace_level = 0;
134 
135  for (;;) {
136  switch (m_tok.kind) {
137  case Token::Kind::LeftParen: {
138  get_next_token();
139  if (!parse_expression(tmpl)) return false;
140  if (m_tok.kind != Token::Kind::RightParen) {
141  inja_throw("parser_error", "unmatched '('");
142  }
143  get_next_token();
144  return true;
145  }
146  case Token::Kind::Id:
147  get_peek_token();
148  if (m_peek_tok.kind == Token::Kind::LeftParen) {
149  // function call, parse arguments
150  Token func_token = m_tok;
151  get_next_token(); // id
152  get_next_token(); // leftParen
153  unsigned int num_args = 0;
154  if (m_tok.kind == Token::Kind::RightParen) {
155  // no args
156  get_next_token();
157  } else {
158  for (;;) {
159  if (!parse_expression(tmpl)) {
160  inja_throw("parser_error", "expected expression, got '" + m_tok.describe() + "'");
161  }
162  num_args += 1;
163  if (m_tok.kind == Token::Kind::RightParen) {
164  get_next_token();
165  break;
166  }
167  if (m_tok.kind != Token::Kind::Comma) {
168  inja_throw("parser_error", "expected ')' or ',', got '" + m_tok.describe() + "'");
169  }
170  get_next_token();
171  }
172  }
173 
174  auto op = m_static.functions.find_builtin(func_token.text, num_args);
175 
176  if (op != Bytecode::Op::Nop) {
177  // swap arguments for default(); see comment in RenderTo()
178  if (op == Bytecode::Op::Default)
179  std::swap(tmpl.bytecodes.back(), *(tmpl.bytecodes.rbegin() + 1));
180  append_function(tmpl, op, num_args);
181  return true;
182  } else {
183  append_callback(tmpl, func_token.text, num_args);
184  return true;
185  }
186  } else if (m_tok.text == "true" || m_tok.text == "false" || m_tok.text == "null") {
187  // true, false, null are json literals
188  if (brace_level == 0 && bracket_level == 0) {
189  json_first = m_tok.text;
190  goto returnJson;
191  }
192  break;
193  } else {
194  // normal literal (json read)
195  tmpl.bytecodes.emplace_back(
196  Bytecode::Op::Push, m_tok.text,
197  m_config.notation == ElementNotation::Pointer ? Bytecode::Flag::ValueLookupPointer : Bytecode::Flag::ValueLookupDot);
198  get_next_token();
199  return true;
200  }
201  // json passthrough
202  case Token::Kind::Number:
203  case Token::Kind::String:
204  if (brace_level == 0 && bracket_level == 0) {
205  json_first = m_tok.text;
206  goto returnJson;
207  }
208  break;
209  case Token::Kind::Comma:
210  case Token::Kind::Colon:
211  if (brace_level == 0 && bracket_level == 0) {
212  inja_throw("parser_error", "unexpected token '" + m_tok.describe() + "'");
213  }
214  break;
215  case Token::Kind::LeftBracket:
216  if (brace_level == 0 && bracket_level == 0) {
217  json_first = m_tok.text;
218  }
219  bracket_level += 1;
220  break;
221  case Token::Kind::LeftBrace:
222  if (brace_level == 0 && bracket_level == 0) {
223  json_first = m_tok.text;
224  }
225  brace_level += 1;
226  break;
227  case Token::Kind::RightBracket:
228  if (bracket_level == 0) {
229  inja_throw("parser_error", "unexpected ']'");
230  }
231  --bracket_level;
232  if (brace_level == 0 && bracket_level == 0) goto returnJson;
233  break;
234  case Token::Kind::RightBrace:
235  if (brace_level == 0) {
236  inja_throw("parser_error", "unexpected '}'");
237  }
238  --brace_level;
239  if (brace_level == 0 && bracket_level == 0) goto returnJson;
240  break;
241  default:
242  if (brace_level != 0) {
243  inja_throw("parser_error", "unmatched '{'");
244  }
245  if (bracket_level != 0) {
246  inja_throw("parser_error", "unmatched '['");
247  }
248  return false;
249  }
250 
251  get_next_token();
252  }
253 
254  returnJson:
255  // bridge across all intermediate tokens
256  nonstd::string_view json_text(json_first.data(), m_tok.text.data() - json_first.data() + m_tok.text.size());
257  tmpl.bytecodes.emplace_back(Bytecode::Op::Push, json::parse(json_text), Bytecode::Flag::ValueImmediate);
258  get_next_token();
259  return true;
260  }

262  bool parse_statement(Template& tmpl, nonstd::string_view path) {
263  if (m_tok.kind != Token::Kind::Id) return false;
264 
265  if (m_tok.text == "if") {
266  get_next_token();
267 
268  // evaluate expression
269  if (!parse_expression(tmpl)) return false;
270 
271  // start a new if block on if stack
272  m_if_stack.emplace_back(tmpl.bytecodes.size());
273 
274  // conditional jump; destination will be filled in by else or endif
275  tmpl.bytecodes.emplace_back(Bytecode::Op::ConditionalJump);
276  } else if (m_tok.text == "endif") {
277  if (m_if_stack.empty()) {
278  inja_throw("parser_error", "endif without matching if");
279  }
280  auto& if_data = m_if_stack.back();
281  get_next_token();
282 
283  // previous conditional jump jumps here
284  if (if_data.prev_cond_jump != std::numeric_limits<unsigned int>::max()) {
285  tmpl.bytecodes[if_data.prev_cond_jump].args = tmpl.bytecodes.size();
286  }
287 
288  // update all previous unconditional jumps to here
289  for (unsigned int i: if_data.uncond_jumps) {
290  tmpl.bytecodes[i].args = tmpl.bytecodes.size();
291  }
292 
293  // pop if stack
294  m_if_stack.pop_back();
295  } else if (m_tok.text == "else") {
296  if (m_if_stack.empty())
297  inja_throw("parser_error", "else without matching if");
298  auto& if_data = m_if_stack.back();
299  get_next_token();
300 
301  // end previous block with unconditional jump to endif; destination will be
302  // filled in by endif
303  if_data.uncond_jumps.push_back(tmpl.bytecodes.size());
304  tmpl.bytecodes.emplace_back(Bytecode::Op::Jump);
305 
306  // previous conditional jump jumps here
307  tmpl.bytecodes[if_data.prev_cond_jump].args = tmpl.bytecodes.size();
308  if_data.prev_cond_jump = std::numeric_limits<unsigned int>::max();
309 
310  // chained else if
311  if (m_tok.kind == Token::Kind::Id && m_tok.text == "if") {
312  get_next_token();
313 
314  // evaluate expression
315  if (!parse_expression(tmpl)) return false;
316 
317  // update "previous jump"
318  if_data.prev_cond_jump = tmpl.bytecodes.size();
319 
320  // conditional jump; destination will be filled in by else or endif
321  tmpl.bytecodes.emplace_back(Bytecode::Op::ConditionalJump);
322  }
323  } else if (m_tok.text == "for") {
324  get_next_token();
325 
326  // options: for a in arr; for a, b in obj
327  if (m_tok.kind != Token::Kind::Id)
328  inja_throw("parser_error", "expected id, got '" + m_tok.describe() + "'");
329  Token value_token = m_tok;
330  get_next_token();
331 
332  Token key_token;
333  if (m_tok.kind == Token::Kind::Comma) {
334  get_next_token();
335  if (m_tok.kind != Token::Kind::Id)
336  inja_throw("parser_error", "expected id, got '" + m_tok.describe() + "'");
337  key_token = std::move(value_token);
338  value_token = m_tok;
339  get_next_token();
340  }
341 
342  if (m_tok.kind != Token::Kind::Id || m_tok.text != "in")
343  inja_throw("parser_error",
344  "expected 'in', got '" + m_tok.describe() + "'");
345  get_next_token();
346 
347  if (!parse_expression(tmpl)) return false;
348 
349  m_loop_stack.push_back(tmpl.bytecodes.size());
350 
351  tmpl.bytecodes.emplace_back(Bytecode::Op::StartLoop);
352  if (!key_token.text.empty()) {
353  tmpl.bytecodes.back().value = key_token.text;
354  }
355  tmpl.bytecodes.back().str = static_cast<std::string>(value_token.text);
356  } else if (m_tok.text == "endfor") {
357  get_next_token();
358  if (m_loop_stack.empty()) {
359  inja_throw("parser_error", "endfor without matching for");
360  }
361 
362  // update loop with EndLoop index (for empty case)
363  tmpl.bytecodes[m_loop_stack.back()].args = tmpl.bytecodes.size();
364 
365  tmpl.bytecodes.emplace_back(Bytecode::Op::EndLoop);
366  tmpl.bytecodes.back().args = m_loop_stack.back() + 1; // loop body
367  m_loop_stack.pop_back();
368  } else if (m_tok.text == "include") {
369  get_next_token();
370 
371  if (m_tok.kind != Token::Kind::String) {
372  inja_throw("parser_error", "expected string, got '" + m_tok.describe() + "'");
373  }
374 
375  // build the relative path
376  json json_name = json::parse(m_tok.text);
377  std::string pathname = static_cast<std::string>(path);
378  pathname += json_name.get_ref<const std::string&>();
379  if (pathname.compare(0, 2, "./") == 0) {
380  pathname.erase(0, 2);
381  }
382  // sys::path::remove_dots(pathname, true, sys::path::Style::posix);
383 
384  Template include_template = parse_template(pathname);
385  m_included_templates.emplace(pathname, include_template);
386 
387  // generate a reference bytecode
388  tmpl.bytecodes.emplace_back(Bytecode::Op::Include, json(pathname), Bytecode::Flag::ValueImmediate);
389 
390  get_next_token();
391  } else {
392  return false;
393  }
394  return true;
395  }

412  void append_callback(Template& tmpl, nonstd::string_view name, unsigned int num_args) {
413  // we can merge with back-to-back push value (not lookup)
414  if (!tmpl.bytecodes.empty()) {
415  Bytecode& last = tmpl.bytecodes.back();
416  if (last.op == Bytecode::Op::Push &&

418  last.op = Bytecode::Op::Callback;

421  return;

430  void parse_into(Template& tmpl, nonstd::string_view path) {
431  m_lexer.start(tmpl.content);
432 
433  for (;;) {
434  get_next_token();
435  switch (m_tok.kind) {
436  case Token::Kind::Eof:
437  if (!m_if_stack.empty()) inja_throw("parser_error", "unmatched if");
438  if (!m_loop_stack.empty()) inja_throw("parser_error", "unmatched for");
439  return;
440  case Token::Kind::Text:
441  tmpl.bytecodes.emplace_back(Bytecode::Op::PrintText, m_tok.text, 0u);
442  break;
443  case Token::Kind::StatementOpen:
444  get_next_token();
445  if (!parse_statement(tmpl, path)) {
446  inja_throw("parser_error", "expected statement, got '" + m_tok.describe() + "'");
447  }
448  if (m_tok.kind != Token::Kind::StatementClose) {
449  inja_throw("parser_error", "expected statement close, got '" + m_tok.describe() + "'");
450  }
451  break;
452  case Token::Kind::LineStatementOpen:
453  get_next_token();
454  parse_statement(tmpl, path);
455  if (m_tok.kind != Token::Kind::LineStatementClose &&
456  m_tok.kind != Token::Kind::Eof) {
457  inja_throw("parser_error", "expected line statement close, got '" + m_tok.describe() + "'");
458  }
459  break;
460  case Token::Kind::ExpressionOpen:
461  get_next_token();
462  if (!parse_expression(tmpl)) {
463  inja_throw("parser_error", "expected expression, got '" + m_tok.describe() + "'");
464  }
465  append_function(tmpl, Bytecode::Op::PrintValue, 1);
466  if (m_tok.kind != Token::Kind::ExpressionClose) {
467  inja_throw("parser_error", "expected expression close, got '" + m_tok.describe() + "'");
468  }
469  break;
470  case Token::Kind::CommentOpen:
471  get_next_token();
472  if (m_tok.kind != Token::Kind::CommentClose) {
473  inja_throw("parser_error", "expected comment close, got '" + m_tok.describe() + "'");
474  }
475  break;
476  default:
477  inja_throw("parser_error", "unexpected token '" + m_tok.describe() + "'");
478  break;
479  }
480  }
