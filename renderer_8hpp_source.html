16 inline nonstd::string_view convert_dot_to_json_pointer(nonstd::string_view dot, std::string& out) {

61  const json* get_imm(const Bytecode& bc) {
62  std::string ptr_buffer;
63  nonstd::string_view ptr;
64  switch (bc.flags & Bytecode::Flag::ValueMask) {
65  case Bytecode::Flag::ValuePop:
66  return nullptr;
67  case Bytecode::Flag::ValueImmediate:
68  return &bc.value;
69  case Bytecode::Flag::ValueLookupDot:
70  ptr = convert_dot_to_json_pointer(bc.str, ptr_buffer);
71  break;
72  case Bytecode::Flag::ValueLookupPointer:
73  ptr_buffer += '/';
74  ptr_buffer += bc.str;
75  ptr = ptr_buffer;
76  break;
77  }
78  try {
79  return &m_data->at(json::json_pointer(ptr.data()));
80  } catch (std::exception&) {
81  // try to evaluate as a no-argument callback
82  if (auto callback = m_callbacks.find_callback(bc.str, 0)) {
83  std::vector<const json*> arguments {};
84  m_tmp_val = callback(arguments);
85  return &m_tmp_val;
86  }
87  inja_throw("render_error", "variable '" + static_cast<std::string>(bc.str) + "' not found");
88  return nullptr;

199  m_stack.emplace_back(std::move(result));
200  break;
201  }
202  case Bytecode::Op::Lower: {
203  auto result = get_args(bc)[0]->get<std::string>();

205  pop_args(bc);

207  break;
208  }

213  pop_args(bc);
214  m_stack.emplace_back(std::move(result));
215  break;
216  }

219  pop_args(bc);

221  break;

226  pop_args(bc);
227  m_stack.emplace_back(std::move(result));
228  break;

232  pop_args(bc);

234  break;

256  break;

262  break;

266  pop_args(bc);
